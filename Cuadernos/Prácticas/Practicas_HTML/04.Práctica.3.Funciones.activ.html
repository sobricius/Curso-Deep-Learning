# 7.- La funci√≥n de activaci√≥n recibida por argumento: crea y compila
def creaModelo ( funcion_activacion='relu' ):
  modelo = Sequential()
  # 7.- A√±adir capas al modelo:
  #        - Capa de entrada: 784 entradas, 512 neuronas, funci√≥n de activaci√≥n recibida por argumento
  modelo.add( Dense(512, activation=funcion_activacion, input_shape=(784,)) )
  #        - Capa oculta: 512 neuronas, funci√≥n de activaci√≥n recibida por argumento
  modelo.add( Dense(10, activation='softmax') )
  #        - Compilaci√≥n:




  # 8.- Compilaci√≥n:
  #        - optimizador: RMSprop
  #        - m√©trica: Exactitud (accuracy)
  #        - funci√≥n de p√©rdidas: deducirla
  modelo.compile(optimizer=RMSprop(), loss='categorical_crossentropy', metrics=['accuracy'])

  return modelo
